int multiplyMatrix(int arr1[n1][m1], int arr2[n2][m2]){
  int arr3[n1][m2] = {0};
  int i, j, k;
  if(m1==n2){
    for(i=0;i<n1;i++){
      for(j=0;j<m2;j++){
        for(k=0;k<m1;k++){
          arr3[i][j] += arr1[i][k] * arr2[k][j]; 
        }
      }
    }
    return arr3[n1][m2];
  }
  else{
    cout << "Matrix Multiplication is not possible";
    return 0;
  }
}
